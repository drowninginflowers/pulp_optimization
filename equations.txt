  warehouses = ["a", "b"]

let destinations be the set of possible destinations for shipments
  destinations = ["a", "b", "c"]

let shipment_count be a matrix of the number of shipments from a warehouse i to destination j
  shipment_counts[i][j] is an integer >= 0, i in warehouses, j in destinations
  total_shipments = sum(shipment_counts[i][j]) for all i in warehouses, j in destinations

let target_distribution be a map of destinations to the integer number of shipments expected to go there
  target_distribution[j] is an int >= 0 for all j in destinations

let shipment_capacity be a matrix of the capactity to ship from warehouse i to destination j
  shipment_capacity[i][j] is an int >= 0

let warehouse_cost be a map from warehouses to the fixed dollar costs of using them for shipments
  warehouse_cost[i] is a double precise float >= 0.00 for all i in warehouses

let shipment_cost be a matrix of the cost per shipment from warehouse i to destination j
  shipment_cost[i][j] is a double precise float > 0.00 for all i in warehouses, j in destinations

let warehouse_usage be a map of warehouses to binary variables indicating usage
  warehouse_usage[i] = 1 if shipment_counts[i][j] > 0 for any j in destinations else 0

  delivery_estimate[i][j] is an integer >= 1 for all i in warehouses, j in destinations

DECISION VARIABLES:
shipment_count, [str][str] = int, matrix of integer decision variables
warehouse_usage, [str] = 0 or 1, map of binary decision variables

DEPENDENT VARIABLES
total_cost, float, objective variable

OBJECTIVE:
minimize total_cost

OBJECTIVE FUNCTION:
total_cost = sum(shipment_count[i][j] * shipment_cost[i][j]) + sum(warehouse_usage[i] * warehouse_cost[i]) across all i and j

WAREHOUSE USAGE CONSTRAINTS:
warehouse_usage[j] = 0 or 1
shipment_count[i][j] <= shipment_capacity[i][j] * warehouse_usage[j] for all i and j

SHIPMENT COUNT CONSTRAINTS:
sum(shipment_count[i][j]) = target_distribution[j] for a fixed j across all i
shipment_count[i][j] >= 0 for all i and j
shipment_count[i][j] <= shipment_capacity[i][j] for all i and j

DELIVERY TARGET CONSTRAINTS:
delivery_target_failure[i][j] = 0 or 1 for all i and j
(delivery_target_failure[i][j] * shipment_count[i][j]) / sum(shipment_count[i][j]) >= 0 for a fixed j across all i
(delivery_target_failure[i][j] * shipment_count[i][j]) / sum(shipment_count[i][j]) <= delivery_tolerance for a fixed j across all i
